#include"Vector.h"


Vector::Vector():x(0),y(0),z(0)
{
	//cout<<"This Is A Default Constructor.";
}

Vector Vector::operator +(const Vector &v)const  //Addition
{
	Vector res;
	res.x = x + v.x;
	res.y = y + v.y;
	res.z = z + v.z;
	return res;
}

Vector Vector::operator -(const Vector &v)const  //Subtraction
{
	Vector res;
	res.x = x - v.x;
	res.y = y - v.y;
	res.z = z - v.z;
	return res;
}

Vector Vector::operator *(const int k)const  //Multiplication By Scalar
{
	Vector res;
	res.x = k * x;
	res.y = k * y;
	res.z = k * z;
	return res;
}

void Vector::GetData()			//Input The Vector
{
	cout<<"
Enter The Value Of x : ";
	cin>>x;
	cout<<"
Enter The Value Of y : ";
	cin>>y;
	cout<<"
Enter The Value Of z : ";
	cin>>z;
}

Vector operator *(int k,const Vector &v)
{
	return v * k;
}

Vector Vector::operator -()const   //Negative Of A Vector
{
	Vector res;
	res.x = -x;
	res.y = -y;
	res.z = -z;
	return res;
}

Vector Vector::operator *(const Vector &v)const //Cross (or Vector)
Product
{
	Vector res;
	res.x = (y * v.z) - (z * v.y);
	res.y = (z * v.x) - (x * v.z);
	res.z = (x * v.y) - (y * v.x);
	return res;
}

void Vector::DotProd(Vector v2)    //Dot(Or Scalar) Product
{
	int v1v2;
	v1v2 = (x * v2.x)+(y * v2.y)+(z * v2.z);
	cout<<"
The Value Of A.B Is: "<<v1v2;
}

ostream & Vector::Show(ostream & out)const
{
	if((x<0) && (y<0) && (z<0))
	out<<x<<"I"<<y<<"J"<<z<<"K";

	if((x<0) && (y<0) && (z>=0))
	out<<x<<"I"<<y<<"J+"<<z<<"K";

	if((x<0) && (y>=0) && (z<0))
	out<<x<<"I+"<<y<<"J"<<z<<"K";

	if((x<0) && (y>=0) && (z>=0))
	out<<x<<"I+"<<y<<"J+"<<z<<"K";

	if((x>=0) && (y<0) && (z<0))
	out<<x<<"I"<<y<<"J"<<z<<"K";

	if((x>=0) && (y<0) && (z>=0))
	out<<x<<"I"<<y<<"J+"<<z<<"K";

	if((x>=0) && (y>=0) && (z<0))
	out<<x<<"I+"<<y<<"J"<<z<<"K";

	if((x>=0) && (y>=0) && (z>=0))
	out<<x<<"I+"<<y<<"J+"<<z<<"K";

	return out;
}
