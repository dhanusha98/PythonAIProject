//function for returning the total stock in store
int cement_in::stock_in()
  {
	int temp=0;
	file.seekg(0L,ios::beg);
	while(file.read((char*)&f,sizeof(f)))
	{
	if(f.flage!='*')
	temp=temp+f.qtt;
	}
	file.clear();
	return(temp);
  }
